399:         u[j].a = p[i].a;                                                \
400:         u[j].b = p[i].b;                                                \
401:       } else if (u[j].a == p[i].a) {                                    \
402:         u[j].b = PetscMin(u[j].b,p[i].b);                               \
403:       }                                                                 \
404:       p[i].a = v.a;                                                     \
405:       p[i].b = v.b;                                                     \
406:     }                                                                   \
407:   }
408: #define DEF_PackPair(type1,type2)                                       \
409:   typedef struct {type1 a; type2 b;} PairType(type1,type2);             \
410:   static void CPPJoin3_(Pack_,type1,type2)(PetscInt n,PetscInt bs,const PetscInt *idx,const void *unpacked,void *packed) { \
411:     const PairType(type1,type2) *u = (const PairType(type1,type2)*)unpacked; \
412:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;          \
413:     PetscInt i;                                                         \
414:     for (i=0; i<n; i++) {                                               \

416:       p[i].b = u[idx[i]].b;                                             \

421:     const PairType(type1,type2) *p = (const PairType(type1,type2)*)packed; \
422:     PetscInt i;                                                         \

438:     PairType(type1,type2) *u = (PairType(type1,type2)*)unpacked;        \
439:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;          \
440:     PetscInt i;                                                         \
441:     for (i=0; i<n; i++) {                                               \
442:       PetscInt j = idx[i];                                              \
443:       PairType(type1,type2) v;                                          \
444:       v.a = u[j].a;                                                     \
445:       v.b = u[j].b;                                                     \
446:       u[j].a = p[i].a;                                                  \
447:       u[j].b = p[i].b;                                                  \
448:       p[i].a = v.a;                                                     \
449:       p[i].b = v.b;                                                     \
450:     }                                                                   \
451:   }                                                                     \
452:   static void FetchAndAdd_ ## type1 ## _ ## type2(PetscInt n,PetscInt bs,const PetscInt *idx,void *unpacked,void *packed) { \
453:     PairType(type1,type2) *u = (PairType(type1,type2)*)unpacked;       \
454:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;         \
455:     PetscInt i;                                                         \
456:     for (i=0; i<n; i++) {                                               \
457:       PetscInt j = idx[i];                                              \
458:       PairType(type1,type2) v;                                          \
459:       v.a = u[j].a;                                                     \
460:       v.b = u[j].b;                                                     \
461:       u[j].a = v.a + p[i].a;                                            \
462:       u[j].b = v.b + p[i].b;                                            \
463:       p[i].a = v.a;                                                     \
464:       p[i].b = v.b;                                                     \
465:     }                                                                   \
466:   }                                                                     \
467:   DEF_UnpackXloc(type1,type2,Max,>)                                     \
468:   DEF_UnpackXloc(type1,type2,Min,<)                                     \
469:   static void CPPJoin3_(PackInit_,type1,type2)(PetscSFBasicPack link) { \

533: static PetscErrorCode PetscSFSetUp_Basic(PetscSF sf)
534: {
535:   PetscSF_Basic *bas = (PetscSF_Basic*)sf->data;
537:   PetscInt *rlengths,*ilengths,i;
538:   MPI_Comm comm;
539:   MPI_Request *rootreqs,*leafreqs;

542:   PetscObjectGetComm((PetscObject)sf,&comm);
543:   PetscObjectGetNewTag((PetscObject)sf,&bas->tag);
544:   /*
545:    * Inform roots about how many leaves and from which ranks
546:    */
547:   PetscMalloc1(sf->nranks,&rlengths);
548:   /* Determine number, sending ranks, and length of incoming  */
549:   for (i=0; i<sf->nranks; i++) {
550:     rlengths[i] = sf->roffset[i+1] - sf->roffset[i]; /* Number of roots referenced by my leaves; for rank sf->ranks[i] */
551:   }
552:   PetscCommBuildTwoSided(comm,1,MPIU_INT,sf->nranks,sf->ranks,rlengths,&bas->niranks,&bas->iranks,(void**)&ilengths);
553:   PetscFree(rlengths);

555:   /* Send leaf identities to roots */
556:   for (i=0,bas->itotal=0; i<bas->niranks; i++) bas->itotal += ilengths[i];
557:   PetscMalloc2(bas->niranks+1,&bas->ioffset,bas->itotal,&bas->irootloc);
558:   PetscMalloc2(bas->niranks,&rootreqs,sf->nranks,&leafreqs);
559:   bas->ioffset[0] = 0;
560:   for (i=0; i<bas->niranks; i++) {
561:     bas->ioffset[i+1] = bas->ioffset[i] + ilengths[i];
562:     MPI_Irecv(bas->irootloc+bas->ioffset[i],ilengths[i],MPIU_INT,bas->iranks[i],bas->tag,comm,&rootreqs[i]);
563:   }
564:   for (i=0; i<sf->nranks; i++) {
565:     PetscMPIInt npoints;
566:     PetscMPIIntCast(sf->roffset[i+1] - sf->roffset[i],&npoints);
567:     MPI_Isend(sf->rremote+sf->roffset[i],npoints,MPIU_INT,sf->ranks[i],bas->tag,comm,&leafreqs[i]);
568:   }
569:   MPI_Waitall(bas->niranks,rootreqs,MPI_STATUSES_IGNORE);
570:   MPI_Waitall(sf->nranks,leafreqs,MPI_STATUSES_IGNORE);
571:   PetscFree(ilengths);
572:   PetscFree2(rootreqs,leafreqs);
573:   return(0);
574: }

578: static PetscErrorCode PetscSFBasicPackTypeSetup(PetscSFBasicPack link,MPI_Datatype unit)
579: {
581:   PetscBool      isInt,isPetscInt,isPetscReal,is2Int,is2PetscInt;
582:   PetscInt       nPetscIntContig,nPetscRealContig;
583: #if defined(PETSC_HAVE_COMPLEX)
584:   PetscBool isPetscComplex;
585:   PetscInt nPetscComplexContig;
586: #endif

589:   MPIPetsc_Type_compare(unit,MPI_INT,&isInt);
590:   MPIPetsc_Type_compare(unit,MPIU_INT,&isPetscInt);
591:   MPIPetsc_Type_compare_contig(unit,MPIU_INT,&nPetscIntContig);
592:   MPIPetsc_Type_compare(unit,MPIU_REAL,&isPetscReal);
593:   MPIPetsc_Type_compare_contig(unit,MPIU_REAL,&nPetscRealContig);
594: #if defined(PETSC_HAVE_COMPLEX)
595:   MPIPetsc_Type_compare(unit,MPIU_COMPLEX,&isPetscComplex);
596:   MPIPetsc_Type_compare_contig(unit,MPIU_COMPLEX,&nPetscComplexContig);
597: #endif
598:   MPIPetsc_Type_compare(unit,MPI_2INT,&is2Int);
599:   MPIPetsc_Type_compare(unit,MPIU_2INT,&is2PetscInt);
600:   link->bs = 1;
601:   if (isInt) {PackInit_int(link); PackInit_Logical_int(link); PackInit_Bitwise_int(link);}
602:   else if (isPetscInt) {PackInit_PetscInt(link); PackInit_Logical_PetscInt(link); PackInit_Bitwise_PetscInt(link);}
603:   else if (isPetscReal) {PackInit_PetscReal(link); PackInit_Logical_PetscReal(link);}
604: #if defined(PETSC_HAVE_COMPLEX)
605:   else if (isPetscComplex) PackInit_PetscComplex_1(link);
606: #endif
607:   else if (is2Int) PackInit_int_int(link);
608:   else if (is2PetscInt) PackInit_PetscInt_PetscInt(link);
609:   else if (nPetscIntContig) {
610:     if (nPetscIntContig%7 == 0) PackInit_PetscInt_7(link);
611:     else if (nPetscIntContig%5 == 0) PackInit_PetscInt_5(link);
612:     else if (nPetscIntContig%4 == 0) PackInit_PetscInt_4(link);
613:     else if (nPetscIntContig%3 == 0) PackInit_PetscInt_3(link);
614:     else if (nPetscIntContig%2 == 0) PackInit_PetscInt_2(link);
615:     else PackInit_PetscInt(link);
616:     link->bs = nPetscIntContig;
617:     link->unitbytes *= nPetscIntContig;
618:   } else if (nPetscRealContig) {
619:     if (nPetscRealContig%7 == 0) PackInit_PetscReal_7(link);
620:     else if (nPetscRealContig%5 == 0) PackInit_PetscReal_5(link);
621:     else if (nPetscRealContig%4 == 0) PackInit_PetscReal_4(link);
622:     else if (nPetscRealContig%3 == 0) PackInit_PetscReal_3(link);
623:     else if (nPetscRealContig%2 == 0) PackInit_PetscReal_2(link);
624:     else PackInit_PetscReal(link);
625:     link->bs = nPetscRealContig;
626:     link->unitbytes *= nPetscRealContig;
627: #if defined(PETSC_HAVE_COMPLEX)
628:   } else if (nPetscComplexContig) {
629:     if (nPetscComplexContig%7 == 0) PackInit_PetscComplex_7(link);
630:     else if (nPetscComplexContig%5 == 0) PackInit_PetscComplex_5(link);
631:     else if (nPetscComplexContig%4 == 0) PackInit_PetscComplex_4(link);
632:     else if (nPetscComplexContig%3 == 0) PackInit_PetscComplex_3(link);
633:     else if (nPetscComplexContig%2 == 0) PackInit_PetscComplex_2(link);
634:     else PackInit_PetscComplex_1(link);
635:     link->bs = nPetscComplexContig;
636:     link->unitbytes *= nPetscComplexContig;
637: #endif
638:   } else {
639:     MPI_Aint lb,bytes;
640:     MPI_Type_get_extent(unit,&lb,&bytes);
641:     if (lb != 0) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_SUP,"Datatype with nonzero lower bound %ld\n",(long)lb);
642:     if (bytes % sizeof(int)) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_SUP,"No support for type size not divisible by %D",sizeof(int));
643:     switch (bytes / sizeof(int)) {
644:     case 1: PackInit_block_int_1(link); break;
645:     case 2: PackInit_block_int_2(link); break;
646:     case 3: PackInit_block_int_3(link); break;
647:     case 4: PackInit_block_int_4(link); break;
648:     case 5: PackInit_block_int_5(link); break;
649:     case 6: PackInit_block_int_6(link); break;
650:     case 7: PackInit_block_int_7(link); break;
651:     case 8: PackInit_block_int_8(link); break;
652:     default: SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"No support for arbitrary block sizes");
653:     }
654:   }
655:   MPI_Type_dup(unit,&link->unit);
656:   return(0);
657: }

756: static PetscErrorCode PetscSFBasicGetPack(PetscSF sf,MPI_Datatype unit,const void *key,PetscSFBasicPack *mylink)
757: {
758:   PetscSF_Basic    *bas = (PetscSF_Basic*)sf->data;
759:   PetscErrorCode   ierr;
760:   PetscSFBasicPack link,*p;
761:   PetscInt         nrootranks,nleafranks;
762:   const PetscInt   *rootoffset,*leafoffset;

765:   /* Look for types in cache */
766:   for (p=&bas->avail; (link=*p); p=&link->next) {
767:     PetscBool match;
768:     MPIPetsc_Type_compare(unit,link->unit,&match);
769:     if (match) {
770:       *p = link->next;          /* Remove from available list */
771:       goto found;
772:     }
773:   }

775:   /* Create new composite types for each send rank */
776:   PetscSFBasicGetRootInfo(sf,&nrootranks,NULL,&rootoffset,NULL);
777:   PetscSFBasicGetLeafInfo(sf,&nleafranks,NULL,&leafoffset,NULL);
778:   PetscNew(&link);
779:   PetscSFBasicPackTypeSetup(link,unit);
780:   PetscCalloc2(rootoffset[nrootranks]*link->unitbytes,&link->root,leafoffset[nleafranks]*link->unitbytes,&link->leaf);
781:   PetscCalloc1(nrootranks+nleafranks,&link->requests);

783: found:
784:   link->key  = key;
785:   link->next = bas->inuse;
786:   bas->inuse = link;

788:   *mylink = link;
789:   return(0);
790: }

900: /* Send from roots to leaves */
901: static PetscErrorCode PetscSFBcastBegin_Basic(PetscSF sf,MPI_Datatype unit,const void *rootdata,void *leafdata)
902: {
903:   PetscSF_Basic     *bas = (PetscSF_Basic*)sf->data;
904:   PetscErrorCode    ierr;
905:   PetscSFBasicPack  link;
906:   PetscInt          i,nrootranks,nleafranks;
907:   const PetscInt    *rootoffset,*leafoffset,*rootloc,*leafloc;
908:   const PetscMPIInt *rootranks,*leafranks;
909:   MPI_Request       *rootreqs,*leafreqs;
910:   size_t            unitbytes;

913:   PetscSFBasicGetRootInfo(sf,&nrootranks,&rootranks,&rootoffset,&rootloc);
914:   PetscSFBasicGetLeafInfo(sf,&nleafranks,&leafranks,&leafoffset,&leafloc);
915:   PetscSFBasicGetPack(sf,unit,rootdata,&link);

917:   unitbytes = link->unitbytes;

919:   PetscSFBasicPackGetReqs(sf,link,&rootreqs,&leafreqs);
920:   /* Eagerly post leaf receives */
921:   for (i=0; i<nleafranks; i++) {
922:     PetscMPIInt n = leafoffset[i+1] - leafoffset[i];
923:     MPI_Irecv(link->leaf+leafoffset[i]*unitbytes,n,unit,leafranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&leafreqs[i]);
924:   }
925:   /* Pack and send root data */
926:   for (i=0; i<nrootranks; i++) {
927:     PetscMPIInt n          = rootoffset[i+1] - rootoffset[i];
928:     void        *packstart = link->root+rootoffset[i]*unitbytes;
929:     (*link->Pack)(n,link->bs,rootloc+rootoffset[i],rootdata,packstart);
930:     MPI_Isend(packstart,n,unit,rootranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&rootreqs[i]);
931:   }
932:   return(0);
933: }

937: PetscErrorCode PetscSFBcastEnd_Basic(PetscSF sf,MPI_Datatype unit,const void *rootdata,void *leafdata)
938: {
939:   PetscErrorCode   ierr;
940:   PetscSFBasicPack link;
941:   PetscInt         i,nleafranks;
942:   const PetscInt   *leafoffset,*leafloc;

945:   PetscSFBasicGetPackInUse(sf,unit,rootdata,PETSC_OWN_POINTER,&link);
946:   PetscSFBasicPackWaitall(sf,link);
947:   PetscSFBasicGetLeafInfo(sf,&nleafranks,NULL,&leafoffset,&leafloc);
948:   for (i=0; i<nleafranks; i++) {
949:     PetscMPIInt n          = leafoffset[i+1] - leafoffset[i];
950:     const void  *packstart = link->leaf+leafoffset[i]*link->unitbytes;
951:     (*link->UnpackInsert)(n,link->bs,leafloc+leafoffset[i],leafdata,packstart);
952:   }
953:   PetscSFBasicReclaimPack(sf,&link);
954:   return(0);
955: }

996: static PetscErrorCode PetscSFReduceEnd_Basic(PetscSF sf,MPI_Datatype unit,const void *leafdata,void *rootdata,MPI_Op op)
997: {
998:   void             (*UnpackOp)(PetscInt,PetscInt,const PetscInt*,void*,const void*);
999:   PetscErrorCode   ierr;
1000:   PetscSFBasicPack link;
1001:   PetscInt         i,nrootranks;
1002:   PetscMPIInt      typesize = -1;
1003:   const PetscInt   *rootoffset,*rootloc;

1006:   PetscSFBasicGetPackInUse(sf,unit,rootdata,PETSC_OWN_POINTER,&link);
1007:   /* This implementation could be changed to unpack as receives arrive, at the cost of non-determinism */
1008:   PetscSFBasicPackWaitall(sf,link);
1009:   PetscSFBasicGetRootInfo(sf,&nrootranks,NULL,&rootoffset,&rootloc);
1010:   PetscSFBasicPackGetUnpackOp(sf,link,op,&UnpackOp);
1011:   if (UnpackOp) {
1012:     typesize = link->unitbytes;
1013:   }
1014:   else {
1015:     MPI_Type_size(unit,&typesize);
1016:   }
1017:   for (i=0; i<nrootranks; i++) {
1018:     PetscMPIInt n   = rootoffset[i+1] - rootoffset[i];
1019:     char *packstart = (char *) link->root+rootoffset[i]*typesize;

1021:     if (UnpackOp) {
1022:       (*UnpackOp)(n,link->bs,rootloc+rootoffset[i],rootdata,(const void *)packstart);
1023:     }
1024: #if PETSC_HAVE_MPI_REDUCE_LOCAL
1025:     else if (n) { /* the op should be defined to operate on the whole datatype, so we ignore link->bs */
1026:       PetscInt j;

1028:       for (j = 0; j < n; j++) {
1029:         MPI_Reduce_local(packstart+j*typesize,((char *) rootdata)+(rootloc[rootoffset[i]+j])*typesize,1,unit,op);
1030:       }
1031:     }
1032: #else
1033:     else {
1034:       SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"No unpacking reduction operation for this MPI_Op");
1035:     }
1036: #endif
1037:   }
1038:   PetscSFBasicReclaimPack(sf,&link);
1039:   return(0);
1040: }

1055: static PetscErrorCode PetscSFFetchAndOpEnd_Basic(PetscSF sf,MPI_Datatype unit,void *rootdata,const void *leafdata,void *leafupdate,MPI_Op op)
1056: {
1057:   PetscSF_Basic     *bas = (PetscSF_Basic*)sf->data;
1058:   void              (*FetchAndOp)(PetscInt,PetscInt,const PetscInt*,void*,void*);
1059:   PetscErrorCode    ierr;
1060:   PetscSFBasicPack  link;
1061:   PetscInt          i,nrootranks,nleafranks;
1062:   const PetscInt    *rootoffset,*leafoffset,*rootloc,*leafloc;
1063:   const PetscMPIInt *rootranks,*leafranks;
1064:   MPI_Request       *rootreqs,*leafreqs;
1065:   size_t            unitbytes;

1068:   PetscSFBasicGetPackInUse(sf,unit,rootdata,PETSC_OWN_POINTER,&link);
1069:   /* This implementation could be changed to unpack as receives arrive, at the cost of non-determinism */
1070:   PetscSFBasicPackWaitall(sf,link);
1071:   unitbytes = link->unitbytes;
1072:   PetscSFBasicGetRootInfo(sf,&nrootranks,&rootranks,&rootoffset,&rootloc);
1073:   PetscSFBasicGetLeafInfo(sf,&nleafranks,&leafranks,&leafoffset,&leafloc);
1074:   PetscSFBasicPackGetReqs(sf,link,&rootreqs,&leafreqs);
1075:   /* Post leaf receives */
1076:   for (i=0; i<nleafranks; i++) {
1077:     PetscMPIInt n = leafoffset[i+1] - leafoffset[i];
1078:     MPI_Irecv(link->leaf+leafoffset[i]*unitbytes,n,unit,leafranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&leafreqs[i]);
1079:   }
1080:   /* Process local fetch-and-op, post root sends */
1081:   PetscSFBasicPackGetFetchAndOp(sf,link,op,&FetchAndOp);
1082:   for (i=0; i<nrootranks; i++) {
1083:     PetscMPIInt n          = rootoffset[i+1] - rootoffset[i];
1084:     void        *packstart = link->root+rootoffset[i]*unitbytes;

1086:     (*FetchAndOp)(n,link->bs,rootloc+rootoffset[i],rootdata,packstart);
1087:     MPI_Isend(packstart,n,unit,rootranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&rootreqs[i]);
1088:   }
1089:   PetscSFBasicPackWaitall(sf,link);
1090:   for (i=0; i<nleafranks; i++) {
1091:     PetscMPIInt n          = leafoffset[i+1] - leafoffset[i];
1092:     const void  *packstart = link->leaf+leafoffset[i]*unitbytes;
1093:     (*link->UnpackInsert)(n,link->bs,leafloc+leafoffset[i],leafupdate,packstart);
1094:   }
1095:   PetscSFBasicReclaimPack(sf,&link);
1096:   return(0);
1097: }
