 57:   VecView(x2, viewer);

103:   if (!equal) {

106:     SETERRQ(PETSC_COMM_SELF,PETSC_ERR_PLIB, "Error in HDF5 viewer");
107:   }

110:   VecDestroy(&x2);
111:   VecDestroy(&y1);
112:   VecDestroy(&y2);
