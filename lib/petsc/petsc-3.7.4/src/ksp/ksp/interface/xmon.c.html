205: PetscErrorCode  KSPMonitorLGTrueResidualNorm(KSP ksp,PetscInt n,PetscReal rnorm,void *ctx)
206: {

208:   PetscReal      x[2],y[2],scnorm;
209:   Vec            resid,work;

215:   if (!n) {PetscDrawLGReset(lg);}
