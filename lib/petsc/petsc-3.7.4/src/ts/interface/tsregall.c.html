 22: /*@C
 23:   TSRegisterAll - Registers all of the timesteppers in the TS package.

 25:   Not Collective

 27:   Input parameter:
 28: . path - The dynamic library path

 30:   Level: advanced

 32: .keywords: TS, timestepper, register, all
 33: .seealso: TSCreate(), TSRegister(), TSRegisterDestroy()
 34: @*/
 35: PetscErrorCode  TSRegisterAll(void)
 36: {

 40:   if (TSRegisterAllCalled) return(0);
 41:   TSRegisterAllCalled = PETSC_TRUE;

 43:   TSRegister(TSEULER,    TSCreate_Euler);
 44:   TSRegister(TSBEULER,   TSCreate_BEuler);
 45:   TSRegister(TSCN,       TSCreate_CN);
 46:   TSRegister(TSPSEUDO,   TSCreate_Pseudo);
 47:   TSRegister(TSGL,       TSCreate_GL);
 48:   TSRegister(TSSSP,      TSCreate_SSP);
 49:   TSRegister(TSTHETA,    TSCreate_Theta);
 50:   TSRegister(TSALPHA,    TSCreate_Alpha);
 51:   TSRegister(TSALPHA2,   TSCreate_Alpha2);
 52: #if defined(PETSC_HAVE_SUNDIALS)
 53:   TSRegister(TSSUNDIALS, TSCreate_Sundials);
 54: #endif
 55:   TSRegister(TSRK,       TSCreate_RK);
 56:   TSRegister(TSARKIMEX,  TSCreate_ARKIMEX);
 57:   TSRegister(TSROSW,     TSCreate_RosW);
 58:   TSRegister(TSEIMEX,    TSCreate_EIMEX);
 59:   TSRegister(TSMIMEX,    TSCreate_Mimex);
 60:   TSRegister(TSBDF,      TSCreate_BDF);
 61:   return(0);
 62: }
