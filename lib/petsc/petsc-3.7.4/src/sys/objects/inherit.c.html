213: {
215:   const char     *subtype = NULL;

219:   if (cbtype == PETSC_FORTRAN_CALLBACK_SUBTYPE) subtype = obj->type_name;
220:   if (!*cid) {PetscFortranCallbackRegister(obj->classid,subtype,cid);}
