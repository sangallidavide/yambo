 34:   if (draw->markertype == PETSC_DRAW_MARKER_CROSS){
 35:     if (draw->ops->coordinatetopixel && draw->ops->pointpixel) {

 39:         (*draw->ops->pointpixel)(draw,i+k,j+k,cl);
