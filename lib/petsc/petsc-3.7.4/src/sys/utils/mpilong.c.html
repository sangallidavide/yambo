 18:     PetscMPIIntCast((i < numchunks-1) ? CHUNKSIZE : cnt - (numchunks-1)*CHUNKSIZE,&icnt);

 22:     else SETERRQ(comm,PETSC_ERR_SUP,"No support for this datatype");
 23:   }
 24:   return(0);
