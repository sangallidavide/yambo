 63: /*--------------------------------------------------------------------*/
 66: int main(int argc,char **argv)
 67: {
 69:   Vec            x, f;               /* solution, function */
 70:   Tao            tao;                /* Tao solver context */
 71:   AppCtx         user;               /* user-defined work context */

 73:    /* Initialize TAO and PETSc */
 74:   PetscInitialize(&argc,&argv,(char *)0,help);

 76:   MPI_Comm_size(MPI_COMM_WORLD,&user.size);
 77:   MPI_Comm_rank(MPI_COMM_WORLD,&user.rank);
 78:   InitializeData(&user);

 80:   /* Run optimization on rank 0 */
 81:   if (user.rank == 0) {
 82:     /* Allocate vectors */
 83:     VecCreateSeq(PETSC_COMM_SELF,NPARAMETERS,&x);
 84:     VecCreateSeq(PETSC_COMM_SELF,NOBSERVATIONS,&f);

 86:     /* TAO code begins here */

 88:     /* Create TAO solver and set desired solution method */
 89:     TaoCreate(PETSC_COMM_SELF,&tao);
 90:     TaoSetType(tao,TAOPOUNDERS);

 92:     /* Set the function and Jacobian routines. */
 93:     FormStartingPoint(x);
 94:     TaoSetInitialVector(tao,x);
 95:     TaoSetSeparableObjectiveRoutine(tao,f,EvaluateFunction,(void*)&user);

 97:     /* Check for any TAO command line arguments */
 98:     TaoSetFromOptions(tao);

100:     /* Perform the Solve */
101:     TaoSolve(tao);

103:     /* Free TAO data structures */
104:     TaoDestroy(&tao);

106:     /* Free PETSc data structures */
107:     VecDestroy(&x);
108:     VecDestroy(&f);
109:     StopWorkers(&user);
110:   } else {
111:     TaskWorker(&user);
112:   }
113:   PetscFinalize();
114:   return 0;
115: }

120: PetscErrorCode EvaluateFunction(Tao tao, Vec X, Vec F, void *ptr)
121: {
122:   AppCtx         *user = (AppCtx *)ptr;
123:   PetscInt       i;
124:   PetscReal      *x,*f;

128:   VecGetArray(X,&x);
129:   VecGetArray(F,&f);
130:   if (user->size == 1) {
131:     /* Single processor */
132:     for (i=0;i<NOBSERVATIONS;i++) {
133:       RunSimulation(x,i,&f[i],user);
134:     }
135:   } else {
136:     /* Multiprocessor master */
137:     PetscMPIInt tag;
138:     PetscInt    finishedtasks,next_task,checkedin;
139:     PetscReal   f_i;
140:     MPI_Status  status;

142:     next_task=0;
143:     finishedtasks=0;
144:     checkedin=0;

146:     while(finishedtasks < NOBSERVATIONS || checkedin < user->size-1) {
147:       MPI_Recv(&f_i,1,MPIU_REAL,MPI_ANY_SOURCE,MPI_ANY_TAG,PETSC_COMM_WORLD,&status);
148:       if (status.MPI_TAG == IDLE_TAG) {
149:         checkedin++;
150:       } else {

152:         tag = status.MPI_TAG;
153:         f[tag] = (PetscReal)f_i;
154:         finishedtasks++;
155:       }

157:       if (next_task<NOBSERVATIONS) {
158:         MPI_Send(x,NPARAMETERS,MPIU_REAL,status.MPI_SOURCE,next_task,PETSC_COMM_WORLD);
159:         next_task++;

161:       } else {
162:         /* Send idle message */
163:         MPI_Send(x,NPARAMETERS,MPIU_REAL,status.MPI_SOURCE,IDLE_TAG,PETSC_COMM_WORLD);
164:       }
165:     }
166:   }
167:   VecRestoreArray(X,&x);
168:   VecRestoreArray(F,&f);
169:   PetscLogFlops(6*NOBSERVATIONS);
170:   return(0);
171: }

418: PetscErrorCode TaskWorker(AppCtx *user)
419: {
420:   PetscReal      x[NPARAMETERS],f = 0.0;
421:   PetscMPIInt    tag=IDLE_TAG;
422:   PetscInt       index;
423:   MPI_Status     status;

427:   /* Send check-in message to master */

429:   MPI_Send(&f,1,MPIU_REAL,0,IDLE_TAG,PETSC_COMM_WORLD);
430:   while (tag != DIE_TAG) {
431:     MPI_Recv(x,NPARAMETERS,MPIU_REAL,0,MPI_ANY_TAG,PETSC_COMM_WORLD,&status);
432:     tag = status.MPI_TAG;
433:     if (tag == IDLE_TAG) {
434:       MPI_Send(&f,1,MPIU_REAL,0,IDLE_TAG,PETSC_COMM_WORLD);
435:     } else if (tag != DIE_TAG) {
436:       index = (PetscInt)tag;
437:       ierr=RunSimulation(x,index,&f,user);
438:       ierr=MPI_Send(&f,1,MPIU_REAL,0,tag,PETSC_COMM_WORLD);
439:     }
440:   }
441:   return(0);
442: }

446: PetscErrorCode RunSimulation(PetscReal *x, PetscInt i, PetscReal*f, AppCtx *user)
447: {
448:   PetscReal *t = user->t;
449:   PetscReal *y = user->y;
450:   *f = y[i] - PetscExpScalar(-x[0]*t[i])/(x[1] + x[2]*t[i]);
451:   return(0);
452: }

456: PetscErrorCode StopWorkers(AppCtx *user)
457: {
458:   PetscInt       checkedin;
459:   MPI_Status     status;
460:   PetscReal      f,x[NPARAMETERS];

464:   checkedin=0;
465:   while(checkedin < user->size-1) {
466:     MPI_Recv(&f,1,MPIU_REAL,MPI_ANY_SOURCE,MPI_ANY_TAG,PETSC_COMM_WORLD,&status);
467:     checkedin++;
468:     PetscMemzero(x,NPARAMETERS*sizeof(PetscReal));
469:     MPI_Send(x,NPARAMETERS,MPIU_REAL,status.MPI_SOURCE,DIE_TAG,PETSC_COMM_WORLD);
470:   }
471:   return(0);
472: }
