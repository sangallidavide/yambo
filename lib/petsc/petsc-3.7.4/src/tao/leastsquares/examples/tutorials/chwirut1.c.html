 62: /*--------------------------------------------------------------------*/
 65: int main(int argc,char **argv)
 66: {
 68:   Vec            x, f;               /* solution, function */
 69:   Mat            J;                  /* Jacobian matrix */
 70:   Tao            tao;                /* Tao solver context */
 71:   PetscInt       i;               /* iteration information */
 72:   PetscReal      hist[100],resid[100];
 73:   PetscInt       lits[100];
 74:   AppCtx         user;               /* user-defined work context */

 76:   PetscInitialize(&argc,&argv,(char *)0,help);

 78:   /* Allocate vectors */
 79:   VecCreateSeq(MPI_COMM_SELF,NPARAMETERS,&x);
 80:   VecCreateSeq(MPI_COMM_SELF,NOBSERVATIONS,&f);

 82:   /* Create the Jacobian matrix. */
 83:   MatCreateSeqDense(MPI_COMM_SELF,NOBSERVATIONS,NPARAMETERS,NULL,&J);

 85:   for (i=0;i<NOBSERVATIONS;i++) user.idm[i] = i;

 87:   for (i=0;i<NPARAMETERS;i++) user.idn[i] = i;

 89:   /* Create TAO solver and set desired solution method */
 90:   TaoCreate(PETSC_COMM_SELF,&tao);
 91:   TaoSetType(tao,TAOPOUNDERS);

 93:  /* Set the function and Jacobian routines. */
 94:   InitializeData(&user);
 95:   FormStartingPoint(x);
 96:   TaoSetInitialVector(tao,x);
 97:   TaoSetSeparableObjectiveRoutine(tao,f,EvaluateFunction,(void*)&user);
 98:   TaoSetJacobianRoutine(tao, J, J, EvaluateJacobian, (void*)&user);

100:   /* Check for any TAO command line arguments */
101:   TaoSetFromOptions(tao);

103:   TaoSetConvergenceHistory(tao,hist,resid,0,lits,100,PETSC_TRUE);
104:   /* Perform the Solve */
105:   TaoSolve(tao);
106:   TaoView(tao,PETSC_VIEWER_STDOUT_SELF);

108:   /* Free TAO data structures */
109:   TaoDestroy(&tao);

111:    /* Free PETSc data structures */
112:   VecDestroy(&x);
113:   VecDestroy(&f);
114:   MatDestroy(&J);

116:   PetscFinalize();
117:   return 0;
118: }

123: PetscErrorCode EvaluateFunction(Tao tao, Vec X, Vec F, void *ptr)
124: {
125:   AppCtx         *user = (AppCtx *)ptr;
126:   PetscInt       i;
127:   PetscReal      *y=user->y,*x,*f,*t=user->t;

131:   VecGetArray(X,&x);
132:   VecGetArray(F,&f);

134:   for (i=0;i<NOBSERVATIONS;i++) {
135:     f[i] = y[i] - PetscExpScalar(-x[0]*t[i])/(x[1] + x[2]*t[i]);
136:   }
137:   VecRestoreArray(X,&x);
138:   VecRestoreArray(F,&f);
139:   PetscLogFlops(6*NOBSERVATIONS);
140:   return(0);
141: }

143: /*------------------------------------------------------------*/
144: /* J[i][j] = df[i]/dt[j] */
147: PetscErrorCode EvaluateJacobian(Tao tao, Vec X, Mat J, Mat Jpre, void *ptr)
148: {
149:   AppCtx         *user = (AppCtx *)ptr;
150:   PetscInt       i;
151:   PetscReal      *x,*t=user->t;
152:   PetscReal      base;

156:   VecGetArray(X,&x);
157:   for (i=0;i<NOBSERVATIONS;i++) {
158:     base = PetscExpScalar(-x[0]*t[i])/(x[1] + x[2]*t[i]);

160:     user->j[i][0] = t[i]*base;
161:     user->j[i][1] = base/(x[1] + x[2]*t[i]);
162:     user->j[i][2] = base*t[i]/(x[1] + x[2]*t[i]);
163:   }

165:   /* Assemble the matrix */
166:   MatSetValues(J,NOBSERVATIONS,user->idm, NPARAMETERS, user->idn,(PetscReal *)user->j,INSERT_VALUES);
167:   MatAssemblyBegin(J,MAT_FINAL_ASSEMBLY);
168:   MatAssemblyEnd(J,MAT_FINAL_ASSEMBLY);

170:   VecRestoreArray(X,&x);
171:   PetscLogFlops(NOBSERVATIONS * 13);
172:   return(0);
173: }
