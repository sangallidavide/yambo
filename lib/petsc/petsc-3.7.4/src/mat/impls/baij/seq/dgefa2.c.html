 76:     j3   = 2*j;

 89:   if (a[6] == 0.0) {
 90:     if (allowzeropivot) {
 92:       PetscInfo1(NULL,"Zero pivot, row %D\n",1);

116:     }
117:   }

120:   k   = 1;
121:   k3  = 2*k;
122:   kp1 = k + 1;
123:   aa  = a + k3;
124:   for (i = kp1; i <= 2; ++i) {
125:     work[i-1] = aa[i];
126:     aa[i]     = 0.0;
127:   }

129:     stmp   = work[j-1];

131:     ay     = &a[k3 + 1];

134:   }
135:   l = ipvt[k-1];
136:   if (l != k) {
137:     ax   = &a[k3 + 1];
138:     ay   = &a[2*l + 1];
139:     stmp = ax[0]; ax[0] = ay[0]; ay[0] = stmp;
140:     stmp = ax[1]; ax[1] = ay[1]; ay[1] = stmp;
141:   }
142:   return(0);
143: }
