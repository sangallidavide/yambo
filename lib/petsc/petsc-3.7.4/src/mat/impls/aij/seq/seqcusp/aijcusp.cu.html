110: PetscErrorCode MatCUSPCopyToGPU(Mat A)
111: {

113:   Mat_SeqAIJCUSP *cuspstruct = (Mat_SeqAIJCUSP*)A->spptr;
114:   Mat_SeqAIJ     *a          = (Mat_SeqAIJ*)A->data;
115:   PetscInt       m           = A->rmap->n,*ii,*ridx;
116:   CUSPMATRIX     *mat;

120:   if (A->valid_GPU_matrix == PETSC_CUSP_UNALLOCATED || A->valid_GPU_matrix == PETSC_CUSP_CPU) {
121:     PetscLogEventBegin(MAT_CUSPCopyToGPU,A,0,0,0);
122:     /*
123:       It may be possible to reuse nonzero structure with new matrix values but
124:       for simplicity and insured correctness we delete and build a new matrix on
125:       the GPU. Likely a very small performance hit.
126:     */
127:     if (cuspstruct->mat) {
128:       try {
129:         if (cuspstruct->format==MAT_CUSP_ELL)
130:           delete (CUSPMATRIXELL *) cuspstruct->mat;
131:         else if (cuspstruct->format==MAT_CUSP_DIA)
132:           delete (CUSPMATRIXDIA *) cuspstruct->mat;
133:         else
134:           delete (CUSPMATRIX *) cuspstruct->mat;
135:       } catch(char *ex) {
136:         SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"CUSP error: %s", ex);
137:       }
138:     }
139:     try {
140:       cuspstruct->nonzerorow=0;
141:       for (int j = 0; j<m; j++) cuspstruct->nonzerorow += ((a->i[j+1]-a->i[j])>0);
142:       if (a->compressedrow.use) {
143:         m    = a->compressedrow.nrows;
144:         ii   = a->compressedrow.i;
145:         ridx = a->compressedrow.rindex;
146:       } else {
147:         /* Forcing compressed row on the GPU */
148:         int k=0;
149:         PetscMalloc1(cuspstruct->nonzerorow+1, &ii);
150:         PetscMalloc1(cuspstruct->nonzerorow, &ridx);
151:         ii[0]=0;
152:         for (int j = 0; j<m; j++) {
153:           if ((a->i[j+1]-a->i[j])>0) {
154:             ii[k]  = a->i[j];
155:             ridx[k]= j;
156:             k++;
157:           }
158:         }
159:         ii[cuspstruct->nonzerorow] = a->nz;
160:         m = cuspstruct->nonzerorow;
161:       }

163:       /* now build matrix */
164:       mat = new CUSPMATRIX;
165:       mat->resize(m,A->cmap->n,a->nz);
166:       mat->row_offsets.assign(ii,ii+m+1);
167:       mat->column_indices.assign(a->j,a->j+a->nz);
168:       mat->values.assign(a->a,a->a+a->nz);

170:       /* convert to other formats if selected */
171:       if (a->compressedrow.use || cuspstruct->format==MAT_CUSP_CSR) {
172:         cuspstruct->mat = mat;
173:         cuspstruct->format = MAT_CUSP_CSR;
174:       } else {
175:         if (cuspstruct->format==MAT_CUSP_ELL) {
176:           CUSPMATRIXELL *ellMat = new CUSPMATRIXELL(*mat);
177:           cuspstruct->mat = ellMat;
178:         } else {
179:           CUSPMATRIXDIA *diaMat = new CUSPMATRIXDIA(*mat);
180:           cuspstruct->mat = diaMat;
181:         }
182:         delete (CUSPMATRIX*) mat;
183:       }

185:       /* assign the compressed row indices */
186:       if (cuspstruct->indices) delete (CUSPINTARRAYGPU*)cuspstruct->indices;
187:       cuspstruct->indices = new CUSPINTARRAYGPU;
188:       cuspstruct->indices->assign(ridx,ridx+m);

190:       /* free the temporaries */
191:       if (!a->compressedrow.use) {
192:         PetscFree(ii);
193:         PetscFree(ridx);
194:       }
195:       if (cuspstruct->tempvec) delete (CUSPARRAY*)cuspstruct->tempvec;
196:       cuspstruct->tempvec = new CUSPARRAY;
197:       cuspstruct->tempvec->resize(m);
198:     } catch(char *ex) {
199:       SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"CUSP error: %s", ex);
200:     }
201:     WaitForGPU();CHKERRCUSP(ierr);

203:     A->valid_GPU_matrix = PETSC_CUSP_BOTH;

205:     PetscLogEventEnd(MAT_CUSPCopyToGPU,A,0,0,0);
206:   }
207:   return(0);
208: }

310: PetscErrorCode MatMult_SeqAIJCUSP(Mat A,Vec xx,Vec yy)
311: {
312:   Mat_SeqAIJ       *a = (Mat_SeqAIJ*)A->data;
313:   PetscErrorCode   ierr;
314:   Mat_SeqAIJCUSP   *cuspstruct = (Mat_SeqAIJCUSP*)A->spptr;
315:   PetscBool        usecprow = a->compressedrow.use;
316:   CUSPARRAY        *xarray=NULL,*yarray=NULL;
317:   static PetscBool cite = PETSC_FALSE;

320:   /*
321:     DM 1/28/2014: As of version 0.4.0 cusp does not handle the case of
322:     zero matrices well.  It produces segfaults on some platforms.
323:     Therefore we manually check for the case of a zero matrix here.
324:   */
325:   if (a->nz == 0) {
326:     return(0);
327:   }
328:   PetscCitationsRegister("@incollection{msk2013,\n  author = {Victor Minden and Barry F. Smith and Matthew G. Knepley},\n  title = {Preliminary Implementation of {PETSc} Using {GPUs}},\n  booktitle = {GPU Solutions to Multi-scale Problems in Science and Engineering},\n  series = {Lecture Notes in Earth System Sciences},\n  editor = {David A. Yuen and Long Wang and Xuebin Chi and Lennart Johnsson and Wei Ge and Yaolin Shi},\n  publisher = {Springer Berlin Heidelberg},\n  pages = {131--140},\n  year = {2013},\n}\n",&cite);
329:   VecCUSPGetArrayRead(xx,&xarray);
330:   VecCUSPGetArrayWrite(yy,&yarray);
331:   try {
332:     if (usecprow) {
333:       /* use compressed row format */
334:       CUSPMATRIX *mat = (CUSPMATRIX*)cuspstruct->mat;
335:       cusp::multiply(*mat,*xarray,*cuspstruct->tempvec);
336:       VecSet_SeqCUSP(yy,0.0);
337:       thrust::copy(cuspstruct->tempvec->begin(),cuspstruct->tempvec->end(),thrust::make_permutation_iterator(yarray->begin(),cuspstruct->indices->begin()));
338:     } else {
339:       /* do not use compressed row format */
340:       if (cuspstruct->format==MAT_CUSP_ELL) {
341:         CUSPMATRIXELL *mat = (CUSPMATRIXELL*)cuspstruct->mat;
342:         cusp::multiply(*mat,*xarray,*yarray);
343:       } else if (cuspstruct->format==MAT_CUSP_DIA) {
344:         CUSPMATRIXDIA *mat = (CUSPMATRIXDIA*)cuspstruct->mat;
345:         cusp::multiply(*mat,*xarray,*yarray);
346:       } else {
347:         CUSPMATRIX *mat = (CUSPMATRIX*)cuspstruct->mat;
348:         cusp::multiply(*mat,*xarray,*yarray);
349:       }
350:     }

352:   } catch (char *ex) {
353:     SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB,"CUSP error: %s", ex);
354:   }

357:   if (!cuspstruct->stream) {
358:     WaitForGPU();CHKERRCUSP(ierr);
359:   }

377: PetscErrorCode MatMultAdd_SeqAIJCUSP(Mat A,Vec xx,Vec yy,Vec zz)
378: {
379:   Mat_SeqAIJ     *a = (Mat_SeqAIJ*)A->data;
381:   PetscBool      usecprow = a->compressedrow.use;
382:   Mat_SeqAIJCUSP *cuspstruct = (Mat_SeqAIJCUSP*)A->spptr;
383:   CUSPARRAY      *xarray     = NULL,*yarray=NULL,*zarray=NULL;

386:   /*
387:     DM 1/28/2014: As of version 0.4.0 cusp does not handle the case of
388:     zero matrices well.  It produces segfaults on some platforms.
389:     Therefore we manually check for the case of a zero matrix here.
390:   */
391:   if (a->nz == 0) {
392:     return(0);
393:   }
394:   try {
395:     VecCopy_SeqCUSP(yy,zz);
396:     VecCUSPGetArrayRead(xx,&xarray);
397:     VecCUSPGetArrayRead(yy,&yarray);
398:     VecCUSPGetArrayWrite(zz,&zarray);

400:     if (usecprow) {
401:       /* use compressed row format */
402:       CUSPMATRIX *mat = (CUSPMATRIX*)cuspstruct->mat;
403:       cusp::multiply(*mat,*xarray,*cuspstruct->tempvec);
404:       thrust::for_each(thrust::make_zip_iterator(thrust::make_tuple(cuspstruct->tempvec->begin(),
405:                                                                     thrust::make_permutation_iterator(zarray->begin(), cuspstruct->indices->begin()))),
406:                        thrust::make_zip_iterator(thrust::make_tuple(cuspstruct->tempvec->end(),
407:                                                                     thrust::make_permutation_iterator(zarray->end(),cuspstruct->indices->end()))),
408:                        VecCUSPPlusEquals());
409:     } else {

411:       if (cuspstruct->format==MAT_CUSP_ELL) {
412:         CUSPMATRIXELL *mat = (CUSPMATRIXELL*)cuspstruct->mat;
413:         cusp::multiply(*mat,*xarray,*cuspstruct->tempvec);
414:       } else if (cuspstruct->format==MAT_CUSP_DIA) {
415:         CUSPMATRIXDIA *mat = (CUSPMATRIXDIA*)cuspstruct->mat;

419:         cusp::multiply(*mat,*xarray,*cuspstruct->tempvec);
420:       }

422:       if (zarray->size() == cuspstruct->indices->size()) {
423:         thrust::for_each(thrust::make_zip_iterator(thrust::make_tuple(cuspstruct->tempvec->begin(),zarray->begin())),

464: /*@
465:    MatCreateSeqAIJCUSP - Creates a sparse matrix in AIJ (compressed row) format
466:    (the default parallel PETSc format).  This matrix will ultimately pushed down
467:    to NVidia GPUs and use the CUSP library for calculations. For good matrix
468:    assembly performance the user should preallocate the matrix storage by setting
469:    the parameter nz (or the array nnz).  By setting these parameters accurately,
470:    performance during matrix assembly can be increased by more than a factor of 50.


473:    Collective on MPI_Comm

475:    Input Parameters:
476: +  comm - MPI communicator, set to PETSC_COMM_SELF
477: .  m - number of rows
478: .  n - number of columns
479: .  nz - number of nonzeros per row (same for all rows)
480: -  nnz - array containing the number of nonzeros in the various rows
481:          (possibly different for each row) or NULL

483:    Output Parameter:
484: .  A - the matrix

486:    It is recommended that one use the MatCreate(), MatSetType() and/or MatSetFromOptions(),
487:    MatXXXXSetPreallocation() paradigm instead of this routine directly.
488:    [MatXXXXSetPreallocation() is, for example, MatSeqAIJSetPreallocation]

490:    Notes:
491:    If nnz is given then nz is ignored

493:    The AIJ format (also called the Yale sparse matrix format or
494:    compressed row storage), is fully compatible with standard Fortran 77
495:    storage.  That is, the stored row and column indices can begin at
496:    either one (as in Fortran) or zero.  See the users' manual for details.

498:    Specify the preallocated storage with either nz or nnz (not both).
499:    Set nz=PETSC_DEFAULT and nnz=NULL for PETSc to control dynamic memory
500:    allocation.  For large problems you MUST preallocate memory or you
501:    will get TERRIBLE performance, see the users' manual chapter on matrices.

503:    By default, this format uses inodes (identical nodes) when possible, to
504:    improve numerical efficiency of matrix-vector products and solves. We
505:    search for consecutive rows with the same nonzero structure, thereby
506:    reusing matrix information to achieve increased efficiency.

508:    Level: intermediate

510: .seealso: MatCreate(), MatCreateAIJ(), MatSetValues(), MatSeqAIJSetColumnIndices(), MatCreateSeqAIJWithArrays(), MatCreateAIJ()

512: @*/
513: PetscErrorCode  MatCreateSeqAIJCUSP(MPI_Comm comm,PetscInt m,PetscInt n,PetscInt nz,const PetscInt nnz[],Mat *A)
514: {

518:   MatCreate(comm,A);
519:   MatSetSizes(*A,m,n,m,n);
520:   MatSetType(*A,MATSEQAIJCUSP);
521:   MatSeqAIJSetPreallocation_SeqAIJ(*A,nz,(PetscInt*)nnz);
522:   return(0);
523: }
