213: PETSC_STATIC_INLINE PetscErrorCode PetscLayoutFindOwner(PetscLayout map,PetscInt idx,PetscInt *owner)
214: {
216:   PetscMPIInt    lo = 0,hi,t;

219:   *owner = -1;                  /* GCC erroneously issues warning about possibly uninitialized use when error condition */
220:   if (!((map->n >= 0) && (map->N >= 0) && (map->range))) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_ARG_WRONGSTATE,"PetscLayoutSetUp() must be called first");
221:   if (idx < 0 || idx > map->N) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_OUTOFRANGE,"Index %D is out of range",idx);
222:   MPI_Comm_size(map->comm,&hi);
223:   while (hi - lo > 1) {
224:     t = lo + (hi - lo) / 2;
225:     if (idx < map->range[t]) hi = t;
226:     else                     lo = t;
227:   }
228:   *owner = lo;
229:   return(0);
230: }

234: /*@C
235:      PetscLayoutFindOwnerIndex - Find the owning rank and the local index for a global index

237:     Not Collective

239:    Input Parameters:
240: +    map   - the layout
241: -    idx   - global index to find the owner of

243:    Output Parameter:
244: +    owner - the owning rank
245: -    lidx  - local index used by the owner for idx

247:    Level: developer

249:     Fortran Notes:
250:       Not available from Fortran

252: @*/
253: PETSC_STATIC_INLINE PetscErrorCode PetscLayoutFindOwnerIndex(PetscLayout map,PetscInt idx,PetscInt *owner, PetscInt *lidx)
254: {
256:   PetscMPIInt    lo = 0,hi,t;

259:   if (!((map->n >= 0) && (map->N >= 0) && (map->range))) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_ARG_WRONGSTATE,"PetscLayoutSetUp() must be called first");
260:   if (idx < 0 || idx > map->N) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_OUTOFRANGE,"Index %D is out of range",idx);
261:   MPI_Comm_size(map->comm,&hi);
262:   while (hi - lo > 1) {
263:     t = lo + (hi - lo) / 2;
264:     if (idx < map->range[t]) hi = t;
265:     else                     lo = t;
266:   }
267:   if (owner) *owner = lo;
268:   if (lidx) *lidx  = idx-map->range[lo];
269:   return(0);
270: }
